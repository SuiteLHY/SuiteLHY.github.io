## SQL

### 建立表\&初始化資料

* 圖書信息表&thinsp;`book`

    ```sql
    CREATE TABLE `book` (
    `id_book` varchar(64) NOT NULL COMMENT '圖書id',
    `name` varchar(200) NOT NULL COMMENT '書名',
    `author` varchar(150) DEFAULT NULL COMMENT '作者',
    `price` decimal(10,2) NOT NULL COMMENT '單價',
    `quantity` decimal(20,0) NOT NULL COMMENT '庫存量',
    PRIMARY KEY (`id_book`)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
    ```

### 解題

```sql
-- 1. 查詢書名為“數據庫原理”的所有圖書號、書名、作者、庫存數量。
SELECT id_book,
  name,
  author,
  quantity
FROM book
WHERE name = '數據庫原理'
;

-- 2. 檢索借閱圖書號為“B5”的讀者姓名。
SELECT
  r.full_name AS reader_full_name
FROM borrow_record br
INNER JOIN reader r ON r.id_reader = br.id_reader
WHERE br.id_reader = 'B5'
;

-- 3. 從借閱表中刪除讀者“張明”所借圖書信息。
-- - - - - - - - - - - - - - - - - - - -
-- 備份資料：
-- INSERT INTO `9`.`borrow_record`(`id_borrow_record`, `id_book`, `id_reader`, `borrow_time`, `return_time`) VALUES ('6', '1', 'O9', '2021-09-04 14:20:06', '2021-09-09 15:34:45');
-- INSERT INTO `9`.`borrow_record`(`id_borrow_record`, `id_book`, `id_reader`, `borrow_time`, `return_time`) VALUES ('7', '2', 'O9', '2021-09-10 14:20:06', '2021-09-13 15:34:45');
DELETE br FROM borrow_record br
RIGHT JOIN reader r ON r.id_reader = br.id_reader
WHERE r.full_name = '張明'
;

-- 4. 找出當前至少借閱了2本圖書的讀者姓名及所在單位，並儲存到新表book1中。
INSERT INTO book1 (`id_book1`, `reader_full_name`, `reader_company`)
SELECT
  REPLACE(UUID(),'-','') AS id_book1,
  t.reader_full_name,
  t.reader_company
FROM (
    SELECT
      r.full_name AS reader_full_name,
      r.company AS reader_company,
      COUNT(br.id_book) AS borrow_book_num
    FROM reader r
    INNER JOIN borrow_record br ON br.id_reader = r.id_reader
    GROUP BY r.id_reader
) t
WHERE t.borrow_book_num > 1
;

-- 5. 建立未還圖書的讀者姓名和工作單位的檢視view_name。
CREATE VIEW view_name AS
SELECT
  r.full_name AS reader_full_name,
  r.company AS reader_company,
  COUNT(br.id_book) AS unreturn_borrow_book_num
FROM reader r
INNER JOIN borrow_record br ON br.id_reader = r.id_reader
WHERE br.return_time IS NULL
GROUP BY r.id_reader
;

```
